Calidad de Código

Karma

  • Correr tests con feedback instantaneo
  • Reproducir tests en todo tipo de browsers.
  • Integración con frameworks de test (jasmine, mocha, etc.)

karma.conf.js

            
            module.exports = function(config) {
              config.set({
                browsers: ['PhantomJS', 'Chrome'],
                frameworks: ['mocha'],
                reporters: ['spec', 'coverage'],
                files: ['test/'],

                coverageReporter: {
                  dir      : './coverage',
                  reporters: [
                    { type: 'lcov', subdir: '.' },
                    { type: 'text-summary' }
                  ]
                }
              });
            };
